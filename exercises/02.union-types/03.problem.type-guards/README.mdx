TypeScript type guards use the special `is` return type to tell the compiler
what type a value is when the predicate returns `true`:

```ts
type Cat = { meow(): void }
type Dog = { bark(): void }
type Pet = Cat | Dog

function isCat(pet: Pet): pet is Cat {
	return 'meow' in pet
}

function speak(pet: Pet) {
	if (isCat(pet)) {
		pet.meow() // âœ… TypeScript knows it's a Cat
	} else {
		pet.bark() // âœ… TypeScript knows it's a Dog
	}
}
```
